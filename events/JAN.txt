country_event = {
	title = "noloc"
	desc = "noloc"
	id = 140400
	
	trigger = {
		tag = JAN
		NOT = {
			has_country_modifier = mcl_happened_modifier
		}
	}
	
	
	option = {
		name = "noloc"
		add_country_modifier = { name = mcl_happened_modifier duration = 360 }
		
		any_country = {
			limit = {
				exists = no
				is_greater_power = no
                capital_scope = { NOT = { province_control_days = 120 } }
                OR = {
                    is_mobilised = yes
                    total_amount_of_ships = 1
                }
			}
			## Audax Validator "." Ignore_NEXT
			annex_to = FROM
		}
	}
}
